```tsx
// src/pages/ComparePickerPage.tsx
import { useState } from "react"
import { useParams, useNavigate } from "react-router"
import { Input } from "@/components/ui/input"
import { Card, CardContent } from "@/components/ui/card"
import { Badge } from "@/components/ui/badge"
import { TOOLS } from "@/data/tools"
import NotFoundPage from "@/pages/NotFoundPage"
import type { Tool } from "@/types"

export default function ComparePickerPage() {
  const { a } = useParams<{ a: string }>()
  const navigate = useNavigate()
  const [query, setQuery] = useState("")

  const origin = TOOLS.find(t => t.slug === a)
  if (!origin) return <NotFoundPage />

  const candidates: Tool[] = TOOLS.filter(t => t.slug !== origin.slug)
  const trimmed = query.trim().toLowerCase()
  const visible = trimmed
    ? candidates.filter(t =>
        t.name.toLowerCase().includes(trimmed) ||
        t.tagline.toLowerCase().includes(trimmed),
      )
    : candidates

  return (
    <div className="container mx-auto p-6">
      <h1 className="text-3xl font-bold mb-2">Compare {origin.name} with...</h1>
      <p className="text-muted-foreground mb-4">Pick a second tool to see them side by side.</p>
      <Input
        type="search"
        placeholder="Filter tools..."
        value={query}
        onChange={e => setQuery(e.target.value)}
        className="mb-6 max-w-md"
        aria-label="Filter candidates"
      />
      {visible.length === 0 ? (
        <div className="text-center py-12">
          <p className="text-lg">No tools match "{query}".</p>
        </div>
      ) : (
        <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-3 gap-4">
          {visible.map(t => (
            <Card
              key={t.slug}
              role="button"
              tabIndex={0}
              onClick={() => navigate(`/compare/${origin.slug}/vs/${t.slug}`)}
              onKeyDown={e => {
                if (e.key === "Enter" || e.key === " ") {
                  e.preventDefault()
                  navigate(`/compare/${origin.slug}/vs/${t.slug}`)
                }
              }}
              className="cursor-pointer hover:bg-accent"
            >
              <CardContent className="p-4 flex items-center gap-3">
                <img src={t.logo} alt={t.name} className="h-10 w-10 rounded" />
                <div className="flex-1 min-w-0">
                  <div className="font-semibold truncate">{t.name}</div>
                  <div className="text-sm text-muted-foreground truncate">{t.tagline}</div>
                </div>
                <Badge variant="outline">{t.category}</Badge>
              </CardContent>
            </Card>
          ))}
        </div>
      )}
    </div>
  )
}
```

**Part A — `src/features/compare/components/CompareTable.tsx`:**

```tsx
import type { Tool } from "@/types"
import { CATEGORIES } from "@/data/categories"
import { cn } from "@/lib/utils"

interface CompareTableProps {
  a: Tool
  b: Tool
}

const categoryName = (slug: string): string =>
  CATEGORIES.find(c => c.slug === slug)?.name ?? slug

export function CompareTable({ a, b }: CompareTableProps) {
  // `match` flag is only meaningful for primitive rows (Pricing, Category, Rating).
  // Feature lists and URL rows are ReactNodes; we never mark them as matching
  // because their content differs structurally even when individual values overlap.
  const rows: Array<{ label: string; left: React.ReactNode; right: React.ReactNode; match: boolean }> = [
    { label: "Pricing", left: a.pricing, right: b.pricing, match: a.pricing === b.pricing },
    { label: "Category", left: categoryName(a.category), right: categoryName(b.category), match: a.category === b.category },
    { label: "Rating", left: a.rating.toFixed(1), right: b.rating.toFixed(1), match: a.rating.toFixed(1) === b.rating.toFixed(1) },
    {
      label: "Key features",
      left: <ul className="list-disc pl-5">{a.features.slice(0, 5).map(f => <li key={f}>{f}</li>)}</ul>,
      right: <ul className="list-disc pl-5">{b.features.slice(0, 5).map(f => <li key={f}>{f}</li>)}</ul>,
      match: false,
    },
    {
      label: "URL",
      left: <a href={a.url} target="_blank" rel="noopener noreferrer" className="underline">{a.url}</a>,
      right: <a href={b.url} target="_blank" rel="noopener noreferrer" className="underline">{b.url}</a>,
      match: false,
    },
  ]

  return (
    <table className="w-full border-collapse">
      <thead>
        <tr>
          <th className="text-left p-2 w-32"></th>
          <th className="text-left p-2">{a.name}</th>
          <th className="text-left p-2">{b.name}</th>
        </tr>
      </thead>
      <tbody>
        {rows.map(row => (
          <tr
            key={row.label}
            data-match={row.match ? "true" : "false"}
            className={cn("border-t", row.match && "opacity-50 text-muted-foreground")}
          >
            <th scope="row" className="text-left p-2 font-medium">{row.label}</th>
            <td className="p-2 align-top">{row.left}</td>
            <td className="p-2 align-top">{row.right}</td>
          </tr>
        ))}
      </tbody>
    </table>
  )
}
```

**COMP-06 implementation note:** Rows where the two tools share an identical primitive value (Pricing, Category, or Rating) get `opacity-50 text-muted-foreground` applied, de-emphasizing the matching row so differences stand out visually. Feature-list and URL rows never trigger the muted state because their content is structural (a `<ul>` of strings or an `<a>` element) — equality on ReactNodes is unreliable, and these rows almost always differ between tools anyway. Tests assert via the `data-match` attribute and the rendered className.

**Part B — `src/pages/ComparePage.tsx`:**

```tsx
import { useEffect } from "react"
import { useParams, useNavigate, Link } from "react-router"
import { ArrowLeftRight, Save } from "lucide-react"
import { Button } from "@/components/ui/button"
import { CompareTable } from "@/features/compare/components/CompareTable"
import { TOOLS } from "@/data/tools"
import { useAuth } from "@/features/auth/hooks/useAuth"
import { useSavedComparisonsStore } from "@/features/compare/store"
import { withToast } from "@/lib/withToast"
import NotFoundPage from "@/pages/NotFoundPage"

export default function ComparePage() {
  const { a, b } = useParams<{ a: string; b: string }>()
  const navigate = useNavigate()
  const { userId } = useAuth()

  const toolA = TOOLS.find(t => t.slug === a)
  const toolB = TOOLS.find(t => t.slug === b)

  // Dev-only assertion: warn if a === b. Render proceeds either way.
  useEffect(() => {
    if (import.meta.env.DEV && a && b && a === b) {
      // eslint-disable-next-line no-console
      console.warn(`[Compare] a === b ("${a}") — comparing a tool to itself.`)
    }
  }, [a, b])

  if (!toolA || !toolB) return <NotFoundPage />

  const onSwap = () => navigate(`/compare/${toolB.slug}/vs/${toolA.slug}`)
  const onChangeLeft = () => navigate(`/compare/${toolB.slug}`)   // keep right, re-pick left
  const onChangeRight = () => navigate(`/compare/${toolA.slug}`)  // keep left, re-pick right

  const onSave = () => {
    if (!userId) return
    withToast(
      () => useSavedComparisonsStore.getState().add(userId, toolA.slug, toolB.slug),
      { success: "Comparison saved" },
    )()
  }

  return (
    <div className="container mx-auto p-6">
      <div className="flex items-center justify-between mb-4 flex-wrap gap-2">
        <h1 className="text-3xl font-bold">Compare</h1>
        <div className="flex items-center gap-2">
          <Button variant="outline" onClick={onSwap}>
            <ArrowLeftRight className="h-4 w-4 mr-2" />Swap
          </Button>
          <Button onClick={onSave} disabled={!userId}>
            <Save className="h-4 w-4 mr-2" />Save Comparison
          </Button>
        </div>
      </div>

      <div className="flex items-center justify-between mb-2">
        <Button variant="ghost" size="sm" onClick={onChangeLeft}>Change left</Button>
        <Button variant="ghost" size="sm" onClick={onChangeRight}>Change right</Button>
      </div>

      <CompareTable a={toolA} b={toolB} />

      <div className="mt-6 text-sm text-muted-foreground">
        Want to compare different tools? <Link to="/categories" className="underline">Browse categories</Link> or use the search bar.
      </div>
    </div>
  )
}
```
