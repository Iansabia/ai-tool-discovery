**Part C — `src/pages/CategoryDetailPage.tsx`:**

```tsx
import { useParams } from "react-router"
import { ToolCard } from "@/features/tools/components/ToolCard"
import { Button } from "@/components/ui/button"
import { Link } from "react-router"
import { CATEGORIES } from "@/data/categories"
import { TOOLS } from "@/data/tools"
import NotFoundPage from "@/pages/NotFoundPage"

export default function CategoryDetailPage() {
  const { slug } = useParams<{ slug: string }>()
  const category = CATEGORIES.find(c => c.slug === slug)
  if (!category) return <NotFoundPage />

  const tools = TOOLS.filter(t => t.category === category.slug)

  return (
    <div className="container mx-auto p-6">
      <h1 className="text-3xl font-bold mb-2">{category.name}</h1>
      <p className="text-muted-foreground mb-6">{category.description}</p>
      {tools.length === 0 ? (
        <div className="text-center py-12">
          <p className="text-lg mb-4">No tools in this category yet.</p>
          <Link to="/categories"><Button>Browse all categories</Button></Link>
        </div>
      ) : (
        <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-3 gap-4">
          {tools.map(tool => <ToolCard key={tool.slug} tool={tool} />)}
        </div>
      )}
    </div>
  )
}
```

**Part A — `src/pages/HomePage.tsx`:**

```tsx
import { Link } from "react-router"
import { Button } from "@/components/ui/button"
import { ToolCard } from "@/features/tools/components/ToolCard"
import { useAuth } from "@/features/auth/hooks/useAuth"
import { useUsersStore } from "@/features/auth/store"
import { TOOLS } from "@/data/tools"
import { CATEGORIES } from "@/data/categories"
import type { CategorySlug } from "@/types"

export default function HomePage() {
  // Subscribe reactively to the user record so interest edits in Phase 4 reflect here.
  const userId = useAuth().userId
  const user = useUsersStore(s => userId ? s.findById(userId) : undefined)
  const interests: CategorySlug[] = user?.interests ?? []

  // Build the recommendation list: every tool whose category is in interests, deduplicated.
  // Show up to 9 to keep the page reasonable.
  const recommendations = TOOLS.filter(t => interests.includes(t.category)).slice(0, 9)

  // For "Recommended because" — match the tool's category against interests; if multiple
  // interests match, show the FIRST (per CONTEXT). Map slug → display name.
  const categoryNameBySlug = new Map(CATEGORIES.map(c => [c.slug, c.name]))

  return (
    <div className="container mx-auto p-6">
      <h1 className="text-3xl font-bold mb-2">Welcome back{user?.displayName ? `, ${user.displayName}` : ""}</h1>
      <p className="text-muted-foreground mb-6">Picks based on your selected interests.</p>

      {interests.length === 0 ? (
        <div className="text-center py-12">
          <p className="text-lg mb-4">You haven't picked any interests yet.</p>
          <Link to="/categories"><Button>Browse Categories</Button></Link>
        </div>
      ) : recommendations.length === 0 ? (
        <div className="text-center py-12">
          <p className="text-lg mb-4">No tools match your interests yet.</p>
          <Link to="/categories"><Button>Browse Categories</Button></Link>
        </div>
      ) : (
        <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-3 gap-4">
          {recommendations.map(tool => (
            <ToolCard
              key={tool.slug}
              tool={tool}
              recommendedBecause={categoryNameBySlug.get(tool.category) ?? tool.category}
            />
          ))}
        </div>
      )}
    </div>
  )
}
```
