**Part A — `src/features/submit/schemas.ts`:**

```typescript
import { z } from "zod"
import type { CategorySlug } from "@/types"

export const SUBMIT_NAME_MAX = 80
export const SUBMIT_DESC_MAX = 300
export const SUBMIT_TAGS_MAX = 5

const ALL_CATEGORY_SLUGS: ReadonlyArray<CategorySlug> = [
  "writing", "coding", "research", "image", "audio",
  "video", "productivity", "design", "data", "marketing",
]

export const submitSchema = z.object({
  name: z.string().min(1, "Name is required").max(SUBMIT_NAME_MAX, `Name must be ${SUBMIT_NAME_MAX} characters or less`),
  url: z.string().min(1, "URL is required").url("Must be a valid URL"),
  category: z.enum(ALL_CATEGORY_SLUGS as unknown as [CategorySlug, ...CategorySlug[]], {
    errorMap: () => ({ message: "Category is required" }),
  }),
  description: z
    .string()
    .min(1, "Description is required")
    .max(SUBMIT_DESC_MAX, `Description must be ${SUBMIT_DESC_MAX} characters or less`),
  /** Raw comma-separated string entered by the user. Validation parses + counts. */
  tagsRaw: z.string().refine(
    (s) => parseTags(s).length <= SUBMIT_TAGS_MAX,
    `Up to ${SUBMIT_TAGS_MAX} tags allowed`,
  ),
})

export type SubmitFormValues = z.infer<typeof submitSchema>

/** Trim, dedup, drop empties. Exported for the form's onSubmit and the test suite. */
export function parseTags(raw: string): string[] {
  return raw
    .split(",")
    .map(s => s.trim())
    .filter(s => s.length > 0)
}
```

**Part B — `src/features/submit/components/SubmitToolForm.tsx`:**

```tsx
import { useForm } from "react-hook-form"
import { zodResolver } from "@hookform/resolvers/zod"
import { useNavigate } from "react-router"
import { Button } from "@/components/ui/button"
import { Input } from "@/components/ui/input"
import { Textarea } from "@/components/ui/textarea"
import {
  Form, FormField, FormItem, FormLabel, FormControl, FormMessage,
} from "@/components/ui/form"
import {
  Select, SelectTrigger, SelectValue, SelectContent, SelectItem,
} from "@/components/ui/select"
import { CATEGORIES } from "@/data/categories"
import { useAuth } from "@/features/auth/hooks/useAuth"
import { useSubmissionStore } from "@/features/submit/store"
import { withToast } from "@/lib/withToast"
import {
  submitSchema, type SubmitFormValues, parseTags, SUBMIT_NAME_MAX, SUBMIT_DESC_MAX,
} from "@/features/submit/schemas"

export function SubmitToolForm() {
  const navigate = useNavigate()
  const { userId } = useAuth()

  const form = useForm<SubmitFormValues>({
    resolver: zodResolver(submitSchema),
    defaultValues: {
      name: "",
      url: "",
      category: undefined as unknown as SubmitFormValues["category"],
      description: "",
      tagsRaw: "",
    },
  })

  const onSubmit = (values: SubmitFormValues) => {
    if (!userId) return
    const tags = parseTags(values.tagsRaw)
    withToast(
      () => useSubmissionStore.getState().add({
        submitterId: userId,
        name: values.name,
        url: values.url,
        category: values.category,
        description: values.description,
        tags,
      }),
      { success: "Tool submitted for review" },
    )()
    navigate("/submit/success", { state: { name: values.name } })
  }

  return (
    <Form {...form}>
      <form onSubmit={form.handleSubmit(onSubmit)} className="space-y-4 max-w-xl">
        <FormField
          control={form.control}
          name="name"
          render={({ field }) => (
            <FormItem>
              <FormLabel>Tool name</FormLabel>
              <FormControl>
                <Input maxLength={SUBMIT_NAME_MAX} placeholder="e.g., Cursor" {...field} />
              </FormControl>
              <FormMessage />
            </FormItem>
          )}
        />
        <FormField
          control={form.control}
          name="url"
          render={({ field }) => (
            <FormItem>
              <FormLabel>URL</FormLabel>
              <FormControl>
                <Input type="url" placeholder="https://..." {...field} />
              </FormControl>
              <FormMessage />
            </FormItem>
          )}
        />
        <FormField
          control={form.control}
          name="category"
          render={({ field }) => (
            <FormItem>
              <FormLabel>Category</FormLabel>
              <Select onValueChange={field.onChange} value={field.value}>
                <FormControl>
                  <SelectTrigger>
                    <SelectValue placeholder="Pick a category" />
                  </SelectTrigger>
                </FormControl>
                <SelectContent>
                  {CATEGORIES.map(c => (
                    <SelectItem key={c.slug} value={c.slug}>{c.name}</SelectItem>
                  ))}
                </SelectContent>
              </Select>
              <FormMessage />
            </FormItem>
          )}
        />
        <FormField
          control={form.control}
          name="description"
          render={({ field }) => (
            <FormItem>
              <FormLabel>Description</FormLabel>
              <FormControl>
                <Textarea maxLength={SUBMIT_DESC_MAX} rows={4} placeholder="What does this tool do?" {...field} />
              </FormControl>
              <FormMessage />
            </FormItem>
          )}
        />
        <FormField
          control={form.control}
          name="tagsRaw"
          render={({ field }) => (
            <FormItem>
              <FormLabel>Tags (optional, comma-separated, max 5)</FormLabel>
              <FormControl>
                <Input placeholder="ai, productivity, scheduling" {...field} />
              </FormControl>
              <FormMessage />
            </FormItem>
          )}
        />
        <div className="flex justify-end">
          <Button type="submit" disabled={!userId}>Submit for review</Button>
        </div>
      </form>
    </Form>
  )
}
```
