**Part A — `src/features/rankings/components/VoteButton.tsx`:**

```tsx
import { ChevronUp, ChevronDown } from "lucide-react"
import { Button } from "@/components/ui/button"
import { cn } from "@/lib/utils"
import { useAuth } from "@/features/auth/hooks/useAuth"
import { useUpvoteStore } from "@/features/rankings/store"
import { withToast } from "@/lib/withToast"
import type { Tool, Vote } from "@/types"

interface VoteButtonProps {
  tool: Tool
}

export function VoteButton({ tool }: VoteButtonProps) {
  const { userId } = useAuth()
  // Subscribe reactively to changes for THIS tool's vote and netCount
  const vote: Vote = useUpvoteStore(s => userId ? s.getVote(userId, tool.slug) : "none")
  const net = useUpvoteStore(s => s.netCount(tool.slug))

  const dispatch = (direction: "up" | "down") => {
    if (!userId) return
    const current = useUpvoteStore.getState().getVote(userId, tool.slug)
    // Compute the resulting message BEFORE dispatching, mirroring the store's state-machine resolution.
    const next: Vote = current === direction ? "none" : direction
    const message =
      next === "up" ? "Voted up" :
      next === "down" ? "Voted down" :
      "Vote removed"
    withToast(
      () => useUpvoteStore.getState().setVote(userId, tool.slug, direction),
      { success: message },
    )()
  }

  return (
    <div className="flex flex-col items-center gap-1">
      <Button
        type="button"
        variant={vote === "up" ? "default" : "outline"}
        size="icon"
        aria-label={`Upvote ${tool.name}`}
        aria-pressed={vote === "up"}
        disabled={!userId}
        onClick={() => dispatch("up")}
        className={cn(vote === "up" && "bg-primary text-primary-foreground")}
      >
        <ChevronUp className="h-4 w-4" />
      </Button>
      <span className="text-sm font-medium tabular-nums" aria-label={`Net votes: ${net}`}>{net}</span>
      <Button
        type="button"
        variant={vote === "down" ? "default" : "outline"}
        size="icon"
        aria-label={`Downvote ${tool.name}`}
        aria-pressed={vote === "down"}
        disabled={!userId}
        onClick={() => dispatch("down")}
        className={cn(vote === "down" && "bg-primary text-primary-foreground")}
      >
        <ChevronDown className="h-4 w-4" />
      </Button>
    </div>
  )
}
```

**Part B — `src/pages/RankingsPage.tsx`:**

```tsx
import { Link } from "react-router"
import { Card, CardContent } from "@/components/ui/card"
import { Badge } from "@/components/ui/badge"
import { TOOLS } from "@/data/tools"
import { useUpvoteStore } from "@/features/rankings/store"
import { VoteButton } from "@/features/rankings/components/VoteButton"

export default function RankingsPage() {
  // Subscribe reactively to data — when any vote changes, the page re-sorts.
  const data = useUpvoteStore(s => s.data)
  const netCount = (slug: string): number => {
    let n = 0
    for (const userMap of Object.values(data)) {
      const v = userMap[slug]
      if (v === "up") n += 1
      else if (v === "down") n -= 1
    }
    return n
  }
  const totalVotes = (slug: string): number => {
    let n = 0
    for (const userMap of Object.values(data)) {
      if (userMap[slug] === "up" || userMap[slug] === "down") n += 1
    }
    return n
  }

  const sorted = [...TOOLS].sort((a, b) => {
    const dn = netCount(b.slug) - netCount(a.slug)
    if (dn !== 0) return dn
    const dt = totalVotes(b.slug) - totalVotes(a.slug)
    if (dt !== 0) return dt
    return a.name.localeCompare(b.name)
  })

  return (
    <div className="container mx-auto p-6">
      <h1 className="text-3xl font-bold mb-2">Rankings</h1>
      <p className="text-muted-foreground mb-6">Cast a vote to see rankings move. Tools are sorted by net upvotes.</p>
      <div className="space-y-2">
        {sorted.map((tool, idx) => (
          <Card key={tool.slug}>
            <CardContent className="p-4 flex items-center gap-4">
              <span className="text-2xl font-bold w-10 text-center text-muted-foreground">#{idx + 1}</span>
              <VoteButton tool={tool} />
              <Link to={`/tools/${tool.slug}`} className="flex items-center gap-3 flex-1 min-w-0">
                <img src={tool.logo} alt={tool.name} className="h-10 w-10 rounded" />
                <div className="flex-1 min-w-0">
                  <div className="font-semibold truncate">{tool.name}</div>
                  <div className="text-sm text-muted-foreground truncate">{tool.tagline}</div>
                </div>
              </Link>
              <Badge variant="outline">{tool.category}</Badge>
            </CardContent>
          </Card>
        ))}
      </div>
    </div>
  )
}
```

```typescript
import { z } from "zod"

export const REVIEW_TITLE_MAX = 80
export const REVIEW_BODY_MAX = 500

export const reviewSchema = z.object({
  rating: z.number().int().min(1, "Rating is required").max(5),
  title: z.string().max(REVIEW_TITLE_MAX, `Title must be ${REVIEW_TITLE_MAX} characters or less`).optional(),
  body: z
    .string()
    .min(1, "Review is required")
    .max(REVIEW_BODY_MAX, `Review must be ${REVIEW_BODY_MAX} characters or less`),
})

export type ReviewFormValues = z.infer<typeof reviewSchema>
```

**Part B — `src/features/reviews/components/WriteReviewDialog.tsx`:**

```tsx
import { useForm } from "react-hook-form"
import { zodResolver } from "@hookform/resolvers/zod"
import { Star } from "lucide-react"
import { Button } from "@/components/ui/button"
import { Input } from "@/components/ui/input"
import { Textarea } from "@/components/ui/textarea"
import {
  Form,
  FormField,
  FormItem,
  FormLabel,
  FormControl,
  FormMessage,
} from "@/components/ui/form"
import { cn } from "@/lib/utils"
import { reviewSchema, type ReviewFormValues } from "@/features/reviews/schemas"
import { useReviewStore } from "@/features/reviews/store"
import { useAuth } from "@/features/auth/hooks/useAuth"
import { withToast } from "@/lib/withToast"
import type { Tool } from "@/types"

interface WriteReviewDialogProps {
  tool: Tool
  onSuccess?: () => void
}

export function WriteReviewDialog({ tool, onSuccess }: WriteReviewDialogProps) {
  const { userId, currentUser, isGuest } = useAuth()
  const form = useForm<ReviewFormValues>({
    resolver: zodResolver(reviewSchema),
    defaultValues: { rating: 0, title: "", body: "" },
  })

  const onSubmit = (values: ReviewFormValues) => {
    if (!userId) return
    const username = isGuest ? "Guest" : (currentUser?.displayName ?? currentUser?.username ?? "User")
    withToast(
      () => useReviewStore.getState().add({
        toolSlug: tool.slug,
        userId,
        username,
        rating: values.rating,
        title: values.title ?? "",
        body: values.body,
      }),
      { success: "Review posted", error: "Couldn't post review" },
    )()
    form.reset({ rating: 0, title: "", body: "" })
    onSuccess?.()
  }

  const rating = form.watch("rating")

  return (
    <Form {...form}>
      <form onSubmit={form.handleSubmit(onSubmit)} className="space-y-4">
        <FormField
          control={form.control}
          name="rating"
          render={({ field }) => (
            <FormItem>
              <FormLabel>Rating</FormLabel>
              <FormControl>
                <div className="flex gap-1" role="radiogroup" aria-label="Star rating">
                  {[1, 2, 3, 4, 5].map(n => (
                    <Button
                      key={n}
                      type="button"
                      variant="ghost"
                      size="icon"
                      role="radio"
                      aria-checked={rating === n}
                      aria-label={`${n} star${n === 1 ? "" : "s"}`}
                      aria-pressed={rating >= n}
                      onClick={() => field.onChange(n)}
                    >
                      <Star className={cn("h-5 w-5", rating >= n && "fill-current")} />
                    </Button>
                  ))}
                </div>
              </FormControl>
              <FormMessage />
            </FormItem>
          )}
        />
        <FormField
          control={form.control}
          name="title"
          render={({ field }) => (
            <FormItem>
              <FormLabel>Title (optional)</FormLabel>
              <FormControl>
                <Input maxLength={80} placeholder="Short summary" {...field} />
              </FormControl>
              <FormMessage />
            </FormItem>
          )}
        />
        <FormField
          control={form.control}
          name="body"
          render={({ field }) => (
            <FormItem>
              <FormLabel>Review</FormLabel>
              <FormControl>
                <Textarea maxLength={500} rows={4} placeholder="Share your experience" {...field} />
              </FormControl>
              <FormMessage />
            </FormItem>
          )}
        />
        <div className="flex justify-end">
          <Button type="submit" disabled={!userId}>Post review</Button>
        </div>
      </form>
    </Form>
  )
}
```
